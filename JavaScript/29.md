### Напишите цикл, который перебирает числа до 100, возвращая “fizz” на числа кратные 3, “buzz” на числа кратные 5 и “fizzbuzz” на числа кратные 3 и 5

~~~~javascript
 for( var i=0; i<=100; i++) {
    if(i % 3 == 0 && i % 5 == 0) {
        console.log('fizbuzz')
    } else if (i%3 == 0) {
        console.log('fizz')
    } else if (i%5 ==0) {
        console.log('buzz')
    } else {
        console.log(i)
    }
  } 
~~~~
